556  PetscScalar integrate(
557  const SPIVec &a,
558  SPIgrid &grid
559  ){
560  if(grid.ytype==UltraS){ // if using UltraSpherical, then it is in Chebyshev coefficients
561  PetscInt ny=grid.y.rows;
562  PetscInt nyx=a.rows;
563  PetscInt ni=nyx/ny;
564  PetscScalar val=0.0;
565  SPIVec atmp(ny,"atmp");
566  for(PetscInt i=0; i<ni; ++i){
567  for(PetscInt j=0; j<ny; ++j){
568  atmp(j,a(ny*i + j,PETSC_TRUE));
569  }
570  atmp();
571  val += integrate_coeffs(atmp,grid.y);
572  }
573  return val;
574  }
575  else if(grid.ytype==Chebyshev){ // if using UltraSpherical, then it is in Chebyshev coefficients
576  PetscInt ny=grid.y.rows;
577  PetscInt nyx=a.rows;
578  PetscInt ni=nyx/ny;
579  PetscScalar val=0.0;
580  SPIVec atmp(ny,"atmp");
581  SPIVec atmp2(ny,"atmp");
582  for(PetscInt i=0; i<ni; ++i){
583  for(PetscInt j=0; j<ny; ++j){
584  atmp(j,a(ny*i + j,PETSC_TRUE));
585  }
586  atmp();

588  atmp2 = ((grid.That)*atmp);
589  val += integrate_coeffs(atmp2,grid.y);
590  }

592  }
593  else{ // otherwise they are physical values, let's integrate using trapezoidal rule
594  PetscInt ny=grid.y.rows;
