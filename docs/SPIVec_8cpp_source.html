795  PetscScalar integrate_coeffs(
796  const SPIVec &a
797  ){
798  PetscInt n=a.rows;
799  PetscInt N=n-1;
800  SPIVec d(n+1,"d");
801  PetscScalar value=0.0,d0=0.0;
802  for(PetscInt i=1; i<=N+1; ++i){
803  if(i==1){
804  value=0.5*(2.0*a(i-1,PETSC_TRUE)-a(i+1,PETSC_TRUE));
805  }
806  else if(i==N+1){
807  value = 0.5/((PetscScalar)i)*(2.0*a(i-1,PETSC_TRUE));
808  }
809  else if(i==N){
810  value = 0.5/((PetscScalar)i)*(1.0*a(i-1,PETSC_TRUE));
811  }
812  else{
813  value = 0.5/((PetscScalar)i)*(1.0*a(i-1,PETSC_TRUE) - a(i+1,PETSC_TRUE));
814  }
815  d(i,value);
816  if((i%2)==0){ // even
817  d0 -= value;
818  }
819  else{ // odd
820  d0 += value;
821  }
822  }
823  d(0,d0);
824  d(); // assemble
825  return sum(d);
